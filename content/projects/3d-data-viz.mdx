// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// OBJECT POOL PATTERN
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
class ParticlePool {
  private pool: THREE.Mesh[] = [];
  private active = new Set<THREE.Mesh>();
  
  acquire(): THREE.Mesh {
    const mesh = this.pool.pop() || this.create();
    this.active.add(mesh);
    return mesh;
  }
  
  release(mesh: THREE.Mesh) {
    mesh.visible = false;
    this.active.delete(mesh);
    this.pool.push(mesh);
  }
  
  dispose() {
    [...this.pool, ...this.active].forEach(mesh => {
      mesh.geometry.dispose();
      (mesh.material as THREE.Material).dispose();
    });
  }
}

// RESULT: Zero memory leaks, stable at 180MB
